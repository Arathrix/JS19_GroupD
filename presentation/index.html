LESS

@base: #888888;

.box {
  color: saturate(@base, 5%);
  border-color: lighten(@base, 30%);
  div { box-shadow: 10px 10px 5px @base; }
}

compiles to:

.box {
  color: #fe33ac;
  border-color: #fdcdea;
}
.box div {
  box-shadow: 10px 10px 5px #888888;
}

Mixins in LESS

.a, #b {
  color: red;
}
.mixin-class {
  .a();
}
.mixin-id {
  #b();
}

compiles to:

.a, #b {
  color: red;
}
.mixin-class {
  color: red;
}
.mixin-id {
  color: red;
}

Mixins II

.my-hover-mixin() {
  &:hover {
    border: 1px solid red;
  }
}
button {
  .my-hover-mixin();
}

compiles to:

button:hover {
  border: 1px solid red;
}
